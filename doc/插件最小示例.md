# 插件最小示例

下面给出一个不依赖平台命令、可直接照着改的最小插件示例。这个示例只做一件事：回显输入文本，并附带当前 agent 与 run 的部分上下文。

## 目录结构

```text
demo_hello_echo/
  plugin.yaml
  runtime.py
```

## plugin.yaml

```yaml
key: demo.hello_echo
name: Demo Hello Echo Plugin
version: v1
plugin_type: toolset
description: Return payload text with selected runtime context.
entrypoint: runtime:create_plugin
workbench_key: demo_hello_echo
tools:
  - hello_echo
permissions:
  - readonly
actions:
  - name: echo
    tool_name: hello_echo
    description: Echo text and optionally include runtime context summary.
    input_schema:
      type: object
      properties:
        text:
          type: string
          description: Text to echo back.
        include_context:
          type: boolean
          description: Append run and agent info from plugin context.
          default: false
      required:
        - text
hot_reload: true
runtime:
  prefix: "[hello]"
```

## runtime.py

```python
from __future__ import annotations

from typing import Any


class HelloEchoPlugin:
    def __init__(self, *, manifest: dict[str, Any], root_path: str):
        self.manifest = manifest
        self.root_path = root_path
        self.runtime = dict(manifest.get("runtime") or {})
        self.prefix = str(self.runtime.get("prefix") or "[hello]").strip() or "[hello]"

    def describe(self) -> dict[str, Any]:
        return {
            "key": self.manifest["key"],
            "name": self.manifest["name"],
            "version": self.manifest["version"],
            "description": self.manifest.get("description", ""),
            "actions": list(self.manifest.get("actions") or []),
            "root_path": self.root_path,
        }

    def health(self) -> dict[str, Any]:
        return {
            "status": "ok",
            "root_path": self.root_path,
        }

    def invoke(self, action: str, payload: dict[str, Any], context: dict[str, Any]) -> dict[str, Any]:
        action_name = str(action or "").strip().lower()
        if action_name != "echo":
            raise ValueError(f"Unsupported action `{action}`.")

        text = str(payload.get("text") or "").strip()
        if not text:
            raise ValueError("`text` is required.")

        result = {
            "ok": True,
            "action": action_name,
            "message": f"{self.prefix} {text}",
        }
        if payload.get("include_context"):
            result["context"] = {
                "run_id": context.get("run_id"),
                "agent_id": context.get("agent_id"),
                "workspace_id": context.get("workspace_id"),
                "project_id": context.get("project_id"),
            }
        return result

    def shutdown(self) -> dict[str, Any]:
        return {"status": "stopping"}


def create_plugin(*, manifest: dict[str, Any], root_path: str) -> HelloEchoPlugin:
    return HelloEchoPlugin(manifest=manifest, root_path=root_path)
```

## 调用效果

假设模型触发 action：

```json
{
  "action": "echo",
  "payload": {
    "text": "hello plugin",
    "include_context": true
  }
}
```

可能返回：

```json
{
  "ok": true,
  "action": "echo",
  "message": "[hello] hello plugin",
  "context": {
    "run_id": "run_xxx",
    "agent_id": "security_agent",
    "workspace_id": "workspace-default",
    "project_id": "project-default"
  }
}
```
